152 inline void GeodesicAlgorithmBase::set_stop_conditions(std::vector<SurfacePoint>* stop_points,
153  double stop_distance)
154 {
155  m_max_propagation_distance = stop_distance;
156 
157  if(!stop_points)
158  {
159  m_stop_vertices.clear();
160  return;
161  }
162 
163  m_stop_vertices.resize(stop_points->size());
164 
165  std::vector<vertex_pointer> possible_vertices;
166  for(unsigned i = 0; i < stop_points->size(); ++i)
167  {
168  SurfacePoint* point = &(*stop_points)[i];
169 
170  possible_vertices.clear();
171  m_mesh->closest_vertices(point, &possible_vertices);
172 
173  vertex_pointer closest_vertex = NULL;
174  double min_distance = 1e100;
175  for(unsigned j = 0; j < possible_vertices.size(); ++j)
176  {
177  double distance = point->distance(possible_vertices[j]);
178  if(distance < min_distance)
179  {
180  min_distance = distance;
181  closest_vertex = possible_vertices[j];
182  }
183  }
184  if(!closest_vertex) {
185  throw geodesic_error("Geodesic error: No closest vertex was found.");
186  }
187 
188  m_stop_vertices[i].first = closest_vertex;
189  m_stop_vertices[i].second = min_distance;
190  }
191 }
