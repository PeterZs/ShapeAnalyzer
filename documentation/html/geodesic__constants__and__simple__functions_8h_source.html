28 inline double cos_from_edges(double const a, //compute the cosine of the angle given the lengths of the edges
29  double const b,
30  double const c)
31 {
32  if(!(a>1e-50)){

39  throw geodesic_error("Geodesic error: Undefined cos.");

41 
42  double result = (b*b + c*c - a*a)/(2.0*b*c);
43  result = std::max(result, -1.0);
44  return std::min(result, 1.0);
